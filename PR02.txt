import java.util.Scanner;

public class MyProgram {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        System.out.print("Enter 9-digit ISBN: ");
        String isbn = input.nextLine();
        // Check input length
        if (isbn.length() != 9) {
            System.out.println("Error: ISBN must be 9 digits.");
            return;
        }
        int sum = 0;
        // Compute weighted sum for first 9 digits
        for (int i = 0; i < 9; i++) {
            char c = isbn.charAt(i);
            if (!Character.isDigit(c)) {
                System.out.println("Error: ISBN must contain only digits.");
                return;
            }
            int digit = c - '0';
            sum += digit * (10 - i);
        }
        // Compute checksum digit
        int checksum = (11 - (sum % 11)) % 11;
        // Build full ISBN
        String fullISBN = isbn;
        if (checksum == 10) {
            fullISBN += "X";
        } else {
            fullISBN += checksum;
        }
        System.out.println("Full 10-digit ISBN: " + fullISBN);
    }
}
