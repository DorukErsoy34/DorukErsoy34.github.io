public class Shape 
{  
        private double x;
        private double y;
        private double side1;
        private double side2;
        private double side3;
        private boolean isFilled;
        
        
        public Shape(double x1,double y1, double s1, double s2,boolean fill){
            this.x = x1;
            this.y = y1;
            this.side1 = s1;
            this.side2 = s2;
            this.isFilled = fill;
        }
        public Shape(double x1, double y1, double s1, double s2, double s3,boolean fill){
            this.x = x1;
            this.y = y1;
            this.side1 = s1;
            this.side2 = s2;
            this.side3 = s3;
            this.isFilled = fill;
        }
        public double getSide1(){
            return side1;
        }
        public double getSide2(){
            return side2;
        }
        public double getSide3(){
            return side3;
        }
        public void changeSide(int a1, int a2){
            side1 = a1;
            side2 = a2;
        }
        public void changeisfilled(boolean b1){
            isFilled = b1;
        }
        public boolean getisfilled(){
            return isFilled;
        }
        public void rectangleY(int a){
            side1*=a;
            side2*=a;
        }
        public void rectangleA(int a){
            side1*=a;
        }
        public void rectangleB(int a){
            side2*=a;
        }
        
}
