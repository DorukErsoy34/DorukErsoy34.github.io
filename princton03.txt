import java.util.Scanner;

public class MyProgram { 
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        System.out.println("Write num x: ");
        int x = input.nextInt();

        System.out.println("Write num y: ");
        int y = input.nextInt();

        System.out.println("Write num z: ");
        int z = input.nextInt();

        System.out.println(isOrdered(x, y, z));
    }

    static boolean isOrdered(int x, int y, int z) {
        return (x < y && y < z) || (x > y && y > z);
    }
}
